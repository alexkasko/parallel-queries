001    package com.alexkasko.springjdbc.parallel.accessor;
002    
003    import com.google.common.collect.ForwardingCollection;
004    import com.google.common.collect.ImmutableList;
005    import org.springframework.jdbc.core.JdbcOperations;
006    import org.springframework.jdbc.core.namedparam.SqlParameterSource;
007    
008    import java.util.Collection;
009    
010    import static com.google.common.base.Preconditions.checkArgument;
011    import static com.google.common.base.Preconditions.checkNotNull;
012    
013    /**
014     * Abstract accessor implementation as a wrapper around provided collection.
015     * Copies provided collection into inner immutable one on creation. Thread-safe.
016     *
017     * @author alexkasko
018     * Date: 6/11/12
019     * @see DataSourceAccessor
020     */
021    public abstract class ImmutableAccessor<T extends JdbcOperations, P extends SqlParameterSource>
022            extends ForwardingCollection<T> implements DataSourceAccessor<T, P> {
023        protected final ImmutableList<T> delegate;
024    
025        /**
026         * Protected constructor, copies provided collection into inner immutable one
027         *
028         * @param collection collection to wrap
029         */
030        protected ImmutableAccessor(Collection<T> collection) {
031            checkNotNull(collection, "Provided collection must be not null");
032            checkArgument(collection.size() > 0, "Provided collection must be not empty");
033            this.delegate = ImmutableList.copyOf(collection);
034        }
035    
036        /**
037         * {@inheritDoc}
038         */
039        @Override
040        protected Collection<T> delegate() {
041            return delegate;
042        }
043    
044        /**
045         * {@inheritDoc}
046         */
047        @Override
048        public String toString() {
049            final StringBuilder sb = new StringBuilder();
050            sb.append("ImmutableAccessor");
051            sb.append("{delegate=").append(delegate);
052            sb.append('}');
053            return sb.toString();
054        }
055    }
